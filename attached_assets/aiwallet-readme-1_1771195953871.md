### NEAR Integration Code

```typescript
// services/near/wallet.ts
import { setupWalletSelector } from '@near-wallet-selector/core';
import { setupMyNearWallet } from '@near-wallet-selector/my-near-wallet';
import { setupMeteorWallet } from '@near-wallet-selector/meteor-wallet';
import { setupHereWallet } from '@near-wallet-selector/here-wallet';
import { setupSender } from '@near-wallet-selector/sender';

export class NearWallet {
  private selector: WalletSelector;
  private wallet: Wallet;
  
  async initialize() {
    this.selector = await setupWalletSelector({
      network: 'mainnet',
      modules: [
        setupMyNearWallet(),
        setupMeteorWallet(),
        setupHereWallet(),
        setupSender(),
      ],
    });
  }
  
  async connect() {
    const modal = setupModal(this.selector, {
      contractId: 'aiwallet.near',
    });
    modal.show();
  }
  
  async sendTokens(receiverId: string, amount: string) {
    const wallet = await this.selector.wallet();
    
    return wallet.signAndSendTransaction({
      receiverId,
      actions: [{
        type: 'Transfer',
        params: {
          deposit: utils.format.parseNearAmount(amount),
        },
      }],
    });
  }
  
  async getBalance(): Promise<string> {
    const account = await this.selector.getAccounts();
    const balance = await account[0].getAccountBalance();
    return utils.format.formatNearAmount(balance.available);
  }
}
```
